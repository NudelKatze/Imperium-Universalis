	cost = 100
	potential = {
		has_dlc = "Rights of Man"
		government = monarchy
	}
	trigger = {
		is_female = no
		has_heir = yes
		is_lesser_in_union = no
		has_regency = no
		is_at_war = yes
	}
	effect = {
		custom_tooltip = sacrifise_heir_aspect_tooltip
		hidden_effect = {
			add_war_exhaustion = -0.2
			add_prestige = 10
			kill_heir = {}
		}
	}
	ai_will_do = {
		factor = 0
		modifier = {
			factor = 0
			NOT = {
				consort_has_personality = infertile_personality
			}
		}
		modifier = {
			factor = 0
			AND = {
				heir_adm = 2
				heir_dip = 2
				heir_mil = 2
			}
		}
		modifier = {
			factor = 2
			is_at_war = yes
			war_exhaustion = 0.4
		}
	}
}
