##### Arabic Cults #####

cult_dushara = {

	global_unrest = -2
	state_maintenance_modifier = -0.1

	allow = {
		OR = {
			has_unlocked_cult = cult_dushara
			AND = {
				religion = arab_religion
				capital_scope = { superregion = arabia_superregion }
				NOT = { tag = HAT }
			}
		}
		NOT = { has_country_modifier = tawheed }
	}

	sprite = 15

	ai_will_do = {
		factor = 1
	}
}
