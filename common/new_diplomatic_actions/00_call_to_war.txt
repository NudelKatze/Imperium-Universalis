call_to_arms_subject = {
	category = influence
	alert_index = 10
	alert_tooltip = abandon_colony_subject_alert_tooltip.1

	require_acceptance = yes
	is_visible = {
		FROM = {
			is_subject_of = ROOT
			special_vassal_type_trigger = yes
		}
	}
	is_allowed = {
		hidden_trigger = {
			FROM = {
				is_subject_of = ROOT
				special_vassal_type_trigger = yes
			}
		}
		NOT = { FROM = { liberty_desire = 70 } }
		is_at_war = yes
	}
	on_accept = {
		hidden_effect = {
			FROM = {
				join_all_offensive_wars_of = ROOT
				join_all_defensive_wars_of = ROOT
			}
		}
	}

	ai_acceptance = {
		add_entry = {
			name = TRUST
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}
			subtract_variable = { which = ai_value value = 70 }
		}
		add_entry = {
			name = OPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = { which = ai_value value = 0.2 }
		}
		add_entry = {
			name = STABILITY
			export_to_variable = {
				variable_name = ai_value
				value = stability
				who = FROM
			}
			multiply_variable = { which = ai_value value = 5 }
		}
		add_entry = {
			name = WAR_EXHAUSTION
			export_to_variable = {
				variable_name = ai_value
				value = war_exhaustion
				who = FROM
			}
			multiply_variable = { which = ai_value value = -4 }
		}
		add_entry = {
			name = IS_BANKRUPT
			limit = { FROM = { is_bankrupt = yes } }
			export_to_variable = {
				which = ai_value
				value = -100
			}
		}
		add_entry = {
			name = IS_IN_LARGE_DEBT
			limit = { FROM = { is_in_large_debt = yes } }
			export_to_variable = {
				which = ai_value
				value = -50
			}
		}
		add_entry = {
			name = BASE_VALUE_CUSTOM
			limit = { FROM = { ai = yes } }
			export_to_variable = {
				which = ai_value
				value = -30
			}
		}
	}

	ai_will_do = {
		OR = {
			NOT = { war_score = 50 }
			is_in_war = { attacker_leader = ROOT }
			is_in_war = { defender_leader = ROOT }
		}
	}
}
