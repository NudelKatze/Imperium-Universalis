#Clear Britannia

country_event = {
	id = clear_regions.201
	title = "clear_regions.201.t"
	desc = "clear_regions.201.d"
	picture = fingersnap_marduk

	is_triggered_only = yes

	trigger = {

	}

	immediate = {
		hidden_effect = {
			set_global_flag = picked_disable_britannia_flag
		}
	}

	option = { #Colonizable disabling
		name = "clear_regions.201.a"
		hidden_effect = {
			britannia_superregion = {
				limit = {
					NOT = { owned_by = ROOT }
				}
				cede_province = XXX
				set_base_manpower = 1
				set_base_production = 0
				set_base_tax = 0
				remove_all_buildings_scripted_effect = yes
			}
		}
	}

	option = { #Complete disabling
		name = "clear_regions.201.b"
		hidden_effect = {
			britannia_superregion = {
				cede_province = XXX
				set_base_manpower = 0
				set_base_production = 0
				set_base_tax = 0
				remove_all_buildings_scripted_effect = yes
			}
		}
	}

	option = { #Country disabling
		name = "clear_regions.201.c"
		custom_tooltip = COUNTRY_TARGETING_DISABLING_TOOLTIP
		hidden_effect = {
			every_country = {
				limit = {
					ai = yes
					capital_scope = { superregion = britannia_superregion }
				}
				if = {
					limit = {
						any_owned_province = { superregion = asia_superregion }
						NOT = { has_global_flag = asia_superregion_compromised_flag }
					}
					set_global_flag = asia_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = mesopotamia_superregion }
						NOT = { has_global_flag = mesopotamia_superregion_compromised_flag }
					}
					set_global_flag = mesopotamia_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = iran_superregion }
						NOT = { has_global_flag = eran_superregion_compromised_flag }
					}
					set_global_flag = eran_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = china_superregion }
						NOT = { has_global_flag = china_superregion_compromised_flag }
					}
					set_global_flag = china_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = macedonia_superregion }
						NOT = { has_global_flag = macedonia_superregion_compromised_flag }
					}
					set_global_flag = macedonia_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = italia_superregion }
						NOT = { has_global_flag = italia_superregion_compromised_flag }
					}
					set_global_flag = italia_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = syria_superregion }
						NOT = { has_global_flag = transeuphratia_superregion_compromised_flag }
					}
					set_global_flag = transeuphratia_superregion_compromised_flag
				}
				if = {
					limit = {
						any_owned_province = { superregion = eastern_africa_superregion }
						NOT = { has_global_flag = libya_orientalis_superregion_compromised_flag }
					}

					cede_province = XXX

			}

		country_event = {
